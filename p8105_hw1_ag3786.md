``` r
hw1_df = tibble(
vec_rsamp = runif(n = 10, min = 0, max = 5),
vec_logical = vec_rsamp >= 2,
vec_char = letters[1:10],
vec_factor = rep(c("Male", "Female"), length = 10)
)
```
